// 3. La fonction de vérification du mot de passe dans la BDD
function verifierMotDePasse(saisieJoueur) {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('MDP_Nouvelle_Partie');
  var vraiMotDePasse = feuille.getRange('A1').getValue();
  
  return (saisieJoueur.toString().trim() === vraiMotDePasse.toString().trim());
}

// Fonction pour créer la partie et générer l'ID unique
function creerNouvellePartie(nomGroupe, mdpGroupe) {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Systeme_Parties');
  
  var donnees = feuille.getDataRange().getValues();
  
  var nouvelID = "";
  var estUnique = false;
  
  // 1. Génération d'un ID unique
  while (!estUnique) {
    // Crée un ID du type "GAME_48152"
    nouvelID = "GAME_" + Math.floor(Math.random() * 100000);
    estUnique = true;
    
    // Vérifie s'il existe déjà dans la colonne A (index 0 du tableau)
    for (var i = 1; i < donnees.length; i++) { 
      if (donnees[i][0] === nouvelID) {
        estUnique = false;
        break; // L'ID existe déjà, on recommence la boucle pour en tirer un autre
      }
    }
  }
  
  // 2. Enregistrement exact selon tes colonnes Google Sheets
  // [Col A, Col B, Col C, Col D, Col E]
  feuille.appendRow([
    nouvelID,     // Colonne A : ID_Partie
    mdpGroupe,    // Colonne B : Mot_De_Passe
    "",           // Colonne C : Liste_ID_Personnage (On laisse vide pour l'instant)
    nomGroupe,    // Colonne D : Nom_Du_Groupe
    "En_cours"    // Colonne E : Statut <--- C'EST ICI QUE ÇA CHANGE
  ]);
  
  // 3. On renvoie l'ID tout neuf au site web (pour ta variable globale)
  return nouvelID;
}

// Fonction pour récupérer toutes les parties avec le statut "En_cours"
function recupererPartiesEnCours() {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Systeme_Parties');
  var donnees = feuille.getDataRange().getValues();
  
  var partiesActives = [];
  
  // On boucle sur les lignes (on commence à 1 pour ignorer la ligne des titres)
  for (var i = 1; i < donnees.length; i++) {
    // Index 0: ID_Partie, Index 3: Nom_Du_Groupe, Index 4: Statut
    if (donnees[i][4] === "En_cours") {
      partiesActives.push({
        id: donnees[i][0],
        nom: donnees[i][3]
      });
    }
  }
  
  return partiesActives;
}

// Fonction pour vérifier le mot de passe d'un groupe spécifique au chargement
function verifierMotDePassePartie(idPartie, mdpSaisi) {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Systeme_Parties');
  var donnees = feuille.getDataRange().getValues();
  
  // On cherche la ligne correspondant à l'ID de la partie
  for (var i = 1; i < donnees.length; i++) {
    if (donnees[i][0] === idPartie) {
      // L'ID correspond ! On vérifie la colonne B (Index 1) qui contient le mot de passe
      return (donnees[i][1].toString().trim() === mdpSaisi.toString().trim());
    }
  }
  return false; // Sécurité : si l'ID n'est pas trouvé, on refuse l'accès
}

// Fonction pour récupérer la date d'Ivalis
function recupererDateEnJeu() {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Date_En_Jeu');

// Fonction pour récupérer la liste des joueurs depuis la base de données

// 1. Lire toutes les instructions
function recupererInstructionsIA() {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Cerveau_IA');
  var donnees = feuille.getDataRange().getValues();
  
  var listeInstructions = [];
  
  // On commence à 1 pour ignorer la ligne des titres
  for (var i = 1; i < donnees.length; i++) {
    if (donnees[i][0]) { // Si un ID_Instruction existe
      listeInstructions.push({
        id: donnees[i][0],      // Colonne A
        titre: donnees[i][1],   // Colonne B
        contenu: donnees[i][2], // Colonne C
        statut: donnees[i][3]   // Colonne D (on/off)
      });
    }
  }
  return listeInstructions;
}

// 2. Basculer le bouton on/off
function basculerStatutInstructionIA(idInstruction, nouveauStatut) {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Cerveau_IA');
  var donnees = feuille.getDataRange().getValues();
  
  for (var i = 1; i < donnees.length; i++) {
    if (donnees[i][0] === idInstruction) {
      feuille.getRange(i + 1, 4).setValue(nouveauStatut); // Met à jour la colonne D
      return true;
    }
  }
  return false;
}

// 3. Créer ou modifier une instruction
function sauvegarderInstructionIA(id, titre, contenu) {
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Cerveau_IA');
  
  // S'il y a déjà un ID, c'est une modification (Double-clic)
  if (id && id !== "") {
    var donnees = feuille.getDataRange().getValues();
    for (var i = 1; i < donnees.length; i++) {
      if (donnees[i][0] === id) {
        feuille.getRange(i + 1, 2).setValue(titre);
        feuille.getRange(i + 1, 3).setValue(contenu);
        return true;
      }
    }
  } 
  // Sinon, c'est une nouvelle création (Bouton +)
  else {
    var nouvelID = "INST_" + Math.floor(Math.random() * 100000);
    // Ajoute : ID, Titre, Contenu, et "on" par défaut
    feuille.appendRow([nouvelID, titre, contenu, "on"]);
    return true;
  }
}
// 4. Supprimer une instruction
function supprimerInstructionIA(idInstruction) {
  if (!idInstruction) return false;
  
  var classeur = SpreadsheetApp.getActiveSpreadsheet();
  var feuille = classeur.getSheetByName('Cerveau_IA');
  var donnees = feuille.getDataRange().getValues();
  
  // Boucle pour trouver la ligne exacte
  for (var i = 1; i < donnees.length; i++) {
    if (donnees[i][0] === idInstruction) {
      // i commence à 1 (index du tableau), donc i+1 correspond au numéro réel de la ligne dans Google Sheets
      feuille.deleteRow(i + 1); 
      return true;
    }
  }
  return false;
}

  // --- VARIABLE GLOBALE (LE BADGE) ---

  function ouvrirModalNouvellePartie() {

    document.getElementById('overlay-modale').style.display = 'none';
    document.getElementById('modale-cle').style.display = 'none';
    document.getElementById('modale-groupe').style.display = 'none';
    document.getElementById('modale-charger').style.display = 'none';
    document.getElementById('modale-mdp-partie').style.display = 'none';
  }

  // Fonction de vérification de la clé (existante)
  function validerCle() {
    const saisie = document.getElementById('saisie-cle').value;
    const btnOuvrir = event.target;
    
    btnOuvrir.innerText = "Vérification..."; 
    
    google.script.run.withSuccessHandler(function(estValide) {
      btnOuvrir.innerText = "Ouvrir";
      
      if (estValide) {
        document.getElementById('modale-cle').style.display = 'none';
        document.getElementById('modale-groupe').style.display = 'block';
      } else {
        document.getElementById('msg-erreur').style.display = 'block';
      }
    }).verifierMotDePasse(saisie);
  }

  // --- FONCTION : CRÉATION DU GROUPE ---
  function validerCreationGroupe() {
    const nomGroupe = document.getElementById('saisie-nom-groupe').value.trim();
    const mdpGroupe = document.getElementById('saisie-mdp-groupe').value.trim();
    const btnCreer = document.getElementById('btn-creer-groupe');

    // 1. Vérification des champs vides
    if (nomGroupe === "" || mdpGroupe === "") {
      alert("Le Grimoire exige un nom de groupe et un mot de passe valides.");
      return;
    }

    // Effet de chargement sur le bouton
    btnCreer.innerText = "Création...";
    btnCreer.disabled = true;

    // 2. Appel de la fonction côté serveur
    google.script.run.withSuccessHandler(function(nouvelID) {
      
      // 3. Enregistrement dans la variable globale !
      window.ID_PARTIE_COURANTE = nouvelID;

      // 4. On ferme les modales
      fermerModales();
      btnCreer.innerText = "Créer";
      btnCreer.disabled = false;

      // 5. La grande bascule : On cache le menu et on affiche le Jeu
      document.getElementById('ecran-menu').style.display = 'none';
      document.getElementById('ecran-jeu').style.display = 'block';
      
      // Petit bonus : On affiche l'ID sur l'écran pour te prouver que ça a marché
      document.getElementById('affichage-id-courant').innerText = "Session active : " + window.ID_PARTIE_COURANTE;

    }).creerNouvellePartie(nomGroupe, mdpGroupe);
  }

  // --- FONCTIONS : CHARGEMENT DE PARTIE ---
  function ouvrirModalChargerPartie() {
    document.getElementById('overlay-modale').style.display = 'block';
    document.getElementById('modale-charger').style.display = 'block';
    
    // On remet l'interface à zéro le temps de charger
    document.getElementById('chargement-parties').style.display = 'block';
    document.getElementById('liste-parties').style.display = 'none';
    document.getElementById('liste-parties').innerHTML = ''; 
    
    // On demande à Code.gs d'aller chercher les parties actives
    google.script.run.withSuccessHandler(afficherListeParties).recupererPartiesEnCours();
  }

  function afficherListeParties(partiesActives) {
    const divChargement = document.getElementById('chargement-parties');
    const ulListe = document.getElementById('liste-parties');
    
    divChargement.style.display = 'none';
    ulListe.style.display = 'block';
    
    if (partiesActives.length === 0) {
      ulListe.innerHTML = '<li style="text-align:center; color: darkred; padding:10px; list-style:none;">Aucune expédition en cours.</li>';
      return;
    }
    
    // On crée un bouton de parchemin pour chaque partie trouvée
    partiesActives.forEach(function(partie) {
      const li = document.createElement('li');
      li.className = 'item-partie';
      
      // L'ID reste bien caché ici dans le code pour lancer la bonne sauvegarde
      li.onclick = function() {
        jouerSonClic();
        // C'EST ICI QUE ÇA A CHANGÉ : On passe par le mot de passe avant de lancer
        demanderMdpPartie(partie.id);
      };
      
      // On n'affiche plus que le nom du groupe !
      li.innerHTML = '<span class="item-nom">' + partie.nom + '</span>';
      ulListe.appendChild(li);
    });
  }

  // --- NOUVELLES FONCTIONS : VÉRIFICATION DU MOT DE PASSE AU CHARGEMENT ---
  function demanderMdpPartie(idChoisi) {
    // On met l'ID de côté
    window.ID_PARTIE_EN_ATTENTE = idChoisi;
    
    // On cache la liste et on affiche la demande de mot de passe
    document.getElementById('modale-charger').style.display = 'none';
    document.getElementById('modale-mdp-partie').style.display = 'block';
    
    // On vide le champ et cache les erreurs par sécurité
    document.getElementById('saisie-mdp-partie').value = '';
    document.getElementById('msg-erreur-partie').style.display = 'none';
  }

  function validerMdpPartie() {
    const saisie = document.getElementById('saisie-mdp-partie').value;
    const btnValider = document.getElementById('btn-valider-mdp-partie');
    const idPartie = window.ID_PARTIE_EN_ATTENTE;
    
    btnValider.innerText = "Vérification...";
    
    // On appelle Google Sheets pour vérifier si le mot de passe correspond à cet ID
    google.script.run.withSuccessHandler(function(estValide) {
      btnValider.innerText = "Déverrouiller";
      
      if (estValide) {
        // C'est le bon mot de passe ! On lance la partie
        document.getElementById('modale-mdp-partie').style.display = 'none';
        lancerPartieChargee(idPartie);
      } else {
        // Mauvais mot de passe
        document.getElementById('msg-erreur-partie').style.display = 'block';
      }
    }).verifierMotDePassePartie(idPartie, saisie);
  }

  // --- LA BASCULE FINALE ---
  function lancerPartieChargee(idChoisi) {
    // 1. On donne le badge (Variable globale)
    window.ID_PARTIE_COURANTE = idChoisi;
    
    // 2. On ferme les menus
    fermerModales();
    document.getElementById('ecran-menu').style.display = 'none';
    
    // 3. On affiche l'écran de jeu
    document.getElementById('ecran-jeu').style.display = 'block';
    document.getElementById('affichage-id-courant').innerText = "Session active : " + window.ID_PARTIE_COURANTE;
  }

    document.getElementById('etape-menu-parametres').style.display = 'none';

  function afficherListeIA(instructions) {

      const estCoche = (inst.statut === "on") ? "checked" : "";

      div.className = 'item-instruction';

  function remplirSelectFactions(factions) {
    const select = document.getElementById('champ-faction');
    select.innerHTML = '<option value="">-- Aucune / Indépendant --</option>'; 
    factions.forEach(f => {
      const opt = document.createElement('option');
      opt.value = f.id; opt.text = f.nom;
      select.appendChild(opt);
    });
  }

  function ouvrirMenuPersonnages() {
    if (typeof fermerParametres === 'function') fermerParametres();
    const overlay = document.getElementById('overlay-personnages');
    const liste = document.getElementById('conteneur-liste-personnages');
    if (liste.classList.contains('ouvert')) { fermerMenuPersonnages(); return; }

    overlay.style.display = 'block'; liste.style.display = 'block';
    setTimeout(() => { overlay.style.opacity = '1'; liste.classList.add('ouvert'); }, 10);

    const idPartieCourante = window.ID_PARTIE_COURANTE; 
    document.getElementById('chargement-persos').style.display = 'block';
    document.getElementById('liste-html-persos').style.display = 'none';

    google.script.run.withSuccessHandler(afficherListePersonnages).recupererPersonnagesDeLaPartie(idPartieCourante);
  }

  function fermerMenuPersonnages() {
    const overlay = document.getElementById('overlay-personnages');
    const liste = document.getElementById('conteneur-liste-personnages');
    overlay.style.opacity = '0'; liste.classList.remove('ouvert'); 
    setTimeout(() => { overlay.style.display = 'none'; liste.style.display = 'none'; }, 600); 
  }

  function afficherListePersonnages(persos) {
    const conteneur = document.getElementById('liste-html-persos');
    conteneur.innerHTML = "";
    if(persos.length === 0) {
      conteneur.innerHTML = "<p style='text-align:center; padding:10px;'>Aucun héros lié à cette partie.</p>";
    } else {
      persos.forEach(p => {
        const div = document.createElement('div');
        div.className = 'item-perso';
        div.ondblclick = function() { jouerSonClic(); ouvrirFichePerso(p.idPersonnage, p.prenom, p.nom, p.couleur); };
        div.innerHTML = `<span>${p.prenom} ${p.nom}</span>`;
        conteneur.appendChild(div);
      });
    }
    document.getElementById('chargement-persos').style.display = 'none';
    conteneur.style.display = 'block';
  }

  function ouvrirFichePerso(idPersonnage, prenomPerso, nomPerso, couleurPerso) {
    const fiche = document.getElementById('fenetre-fiche-perso');
    const btnSupprimer = document.getElementById('btn-supprimer-perso');
    const btnSauvegarder = document.getElementById('btn-sauvegarder-perso');
    
    document.getElementById('champ-id-personnage').value = "";
    document.getElementById('champ-statut-personnage').value = "Vivant";
    document.getElementById('champ-id-personnage').setAttribute('data-url', "");
    
    document.getElementById('image-portrait-perso').src = "";
    document.getElementById('image-portrait-perso').style.display = 'none';
    document.getElementById('texte-aucun-portrait').style.display = 'block';

    const inputs = fiche.querySelectorAll('.input-perso');
    inputs.forEach(input => { if(input.type === "text") input.value = ""; });
    
    document.getElementById('champ-race').value = ""; // AJOUT ICI
    document.getElementById('champ-genre').value = "";
    document.getElementById('champ-corpulence').value = "";
    document.getElementById('champ-taille').value = "";
    document.getElementById('champ-faction').value = "";
    
    const couleurParDefaut = "#2a1a0f";
    document.getElementById('champ-couleur-token').value = couleurParDefaut;
    appliquerCouleurTheme(couleurParDefaut);

    if (idPersonnage) {
      document.getElementById('titre-nom-personnage').innerText = prenomPerso + " " + nomPerso;
      btnSupprimer.style.display = 'block';
      document.getElementById('champ-id-personnage').value = idPersonnage;
      
      if (couleurPerso) {
        document.getElementById('champ-couleur-token').value = couleurPerso;
        appliquerCouleurTheme(couleurPerso);
      }

      const cleCache = 'ivalis_perso_' + idPersonnage;
      const memoireLocale = localStorage.getItem(cleCache);
      
      if (memoireLocale) {
        remplirFormulairePerso(JSON.parse(memoireLocale));
        btnSauvegarder.innerText = "Synchronisation...";
        btnSauvegarder.style.pointerEvents = "none";
      } else {
        btnSauvegarder.innerText = "Chargement...";
        btnSauvegarder.style.pointerEvents = "none";
      }

      google.script.run
        .withSuccessHandler(function(donneesServeur) {
          if (donneesServeur) {
            localStorage.setItem(cleCache, JSON.stringify(donneesServeur));
            remplirFormulairePerso(donneesServeur);
          }
        })
        .recupererDetailsPersonnage(idPersonnage);

    } else {
      document.getElementById('titre-nom-personnage').innerText = "Nouveau Personnage";
      btnSupprimer.style.display = 'none';
      btnSauvegarder.innerText = "Enregistrer";
      btnSauvegarder.style.pointerEvents = "auto";
    }
    
    fiche.style.display = 'flex';
    const fenetreLargeur = fiche.offsetWidth;
    const fenetreHauteur = fiche.offsetHeight;
    fiche.style.left = (window.innerWidth / 2 - fenetreLargeur / 2) + "px";
    fiche.style.top = (window.innerHeight / 2 - fenetreHauteur / 2) + "px";
  }

function sauvegarderFichePersonnage(donnees) {
  var classeur = SpreadsheetApp.openById(ID_FICHIER_GAMEPLAY);
  var feuille = classeur.getSheetByName('Descriptif_Personnage');
  
  var estNouveau = false;
  var idPersonnage = donnees.idPersonnage;
  
  if (!idPersonnage || idPersonnage === "") {
    estNouveau = true;
    idPersonnage = "PERSO_" + Math.floor(Math.random() * 1000000);
  }
  
  // NOUVEAU : On génère TOUJOURS une nouvelle image à chaque enregistrement
  donnees.urlCloudinary = genererEtStockerPortrait(donnees); 
  
  var ligneData = [
    donnees.idPartie,       // A
    donnees.idJoueur,       // B
    idPersonnage,           // C
    donnees.couleur,        // D
    donnees.prenom,         // E
    donnees.nom,            // F
    donnees.race || "",     // G 
    donnees.urlCloudinary,  // H <-- L'URL est bien placée en H (Index 7)
    estNouveau ? "Vivant" : donnees.statut, // I <-- Statut en I (Index 8)
    donnees.age,            // J
    donnees.genre,          // K
    donnees.corpulence,     // L
    donnees.taille,         // M
    donnees.peau,           // N
    donnees.cheveux,        // O
    donnees.yeux,           // P
    donnees.pilosite,       // Q
    donnees.style,          // R
    donnees.couleursDom,    // S
    donnees.equipement,     // T
    donnees.signes,         // U
    donnees.expression,     // V
    donnees.idFaction       // W
  ];
  
  if (estNouveau) {
    feuille.appendRow(ligneData);
  } else {
    var data = feuille.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][2] === idPersonnage) { 
        feuille.getRange(i + 1, 1, 1, ligneData.length).setValues([ligneData]);
        break;
      }
    }
  }
  
  return {
    id: idPersonnage,
    url: donnees.urlCloudinary
  }; 
}

function genererEtStockerPortrait(donnees) {
  Logger.log("=== DÉBUT DE LA GÉNÉRATION DALL-E 3 ===");
  
  // 1. Récupération de l'instruction
  var classeurIvalis = SpreadsheetApp.openById(ID_FICHIER_IVALIS_BDD);
  var feuilleInst = classeurIvalis.getSheetByName('Cerveau_IA'); 
  
  if (!feuilleInst) {
    Logger.log("🚨 ERREUR : Onglet 'Cerveau_IA' introuvable.");
    return ""; 
  }
  
  var dataInst = feuilleInst.getDataRange().getValues();
  var instructionSupplementaire = "";
  for (var i = 1; i < dataInst.length; i++) {
    if (dataInst[i][0] === "INST_76839") {
      instructionSupplementaire = dataInst[i][2]; 
      break;
    }
  }

  // 2. Construction du Prompt "Naturel" pour DALL-E 3
  var promptOpenAI = "Crée un digital painting croquis pour un portrait de personnage de jeu de rôle heroic fantasy. " +
  "L'esthétique globale doit être avec un éclairage dramatique mais lumineux, des coups de pinceau gestuels et des textures très tactiles sur les matériaux, comme un dessin préparatoire. " +
  "Ne dessine absolument aucun texte ou lettrage sur l'image.\n\n" +
  "Description du personnage :\n" +
  "Il s'agit d'un héros de genre " + donnees.genre + ", ayant environ " + donnees.age + " ans. " +
  "Sa corpulence est " + donnees.corpulence + " et sa taille est " + donnees.taille + ". " +
  "Son teint de peau est " + donnees.peau + ". Ses cheveux sont " + donnees.cheveux + " et il a les yeux " + donnees.yeux + ". " +
  "Pilosité faciale : " + donnees.pilosite + ". " +
  "Son visage porte l'expression suivante : " + donnees.expression + ", et on remarque ces signes distinctifs : " + donnees.signes + ".\n" +
  "Il est vêtu ainsi : " + donnees.style + " avec une palette de couleurs dominantes " + donnees.couleursDom + ". " +
  "Il porte l'équipement visible suivant : " + donnees.equipement + ".\n\n" +
  "Directives de lore et de style additionnelles : " + instructionSupplementaire;

  Logger.log("📝 Prompt DALL-E 3 prêt. Envoi à OpenAI...");

  // 3. Appel à l'API OpenAI (DALL-E 3)
  var urlOpenAI = "https://api.openai.com/v1/images/generations";
  
var payloadOpenAI = {
    "model": "gpt-image-2", 
    "prompt": promptOpenAI,
    "output_format": "webp",
    "n": 1,
    "size": "1024x1792", // Format portrait officiel pris en charge par ce modèle
    "quality": "low" // Le niveau le plus économique possible chez OpenAI
  };
  
  var optionsOpenAI = {
    "method": "post",
    "contentType": "application/json",
    "headers": { 
      "Authorization": "Bearer " + OPENAI_API_KEY 
    },
    "payload": JSON.stringify(payloadOpenAI),
    "muteHttpExceptions": true
  };

  // --- LE SYSTÈME BÉLIER (Anti-Spam 1015) AMÉLIORÉ ---
  var responseOpenAI;
  var tentative = 0;
  var succes = false;
  var texteReponseOpenAI = "";
  var delais = [5000, 15000, 30000]; // Attente : 5s, puis 15s, puis 30s

  while (tentative < 3 && !succes) {
    responseOpenAI = UrlFetchApp.fetch(urlOpenAI, optionsOpenAI);
    texteReponseOpenAI = responseOpenAI.getContentText();
    
    if (texteReponseOpenAI.includes("error code: 1015") || texteReponseOpenAI.includes("Rate Limited")) {
      Logger.log("🛡️ Bloqué par Cloudflare (Tentative " + (tentative + 1) + "/3). Attente de " + (delais[tentative]/1000) + "s avant de retenter...");
      Utilities.sleep(delais[tentative]);
      tentative++;
    } else {
      succes = true; 
    }
  }
  
  // --- LECTURE DE LA RÉPONSE OPENAI ---
  var jsonOpenAI;
  try {
    jsonOpenAI = JSON.parse(texteReponseOpenAI);
  } catch (erreur) {
    Logger.log("🚨 ÉCHEC DÉFINITIF JSON. Réponse brute : " + texteReponseOpenAI);
    return donnees.urlCloudinary || "";
  }
  
  if (jsonOpenAI.error) {
    Logger.log("🚨 ERREUR OPENAI : " + jsonOpenAI.error.message);
    return donnees.urlCloudinary || "";
  }
  
  if (!jsonOpenAI.data || jsonOpenAI.data.length === 0) {
    Logger.log("🚨 ERREUR OPENAI : Aucune image renvoyée.");
    return donnees.urlCloudinary || "";
  }
  
  // --- CRÉATION DE LA FAMEUSE VARIABLE imageSource ---
  var imageSource = jsonOpenAI.data[0].url;
  if (!imageSource && jsonOpenAI.data[0].b64_json) {
    imageSource = "data:image/png;base64," + jsonOpenAI.data[0].b64_json;
  }

  Logger.log("✅ Image générée par OpenAI avec succès ! URL temporaire : " + imageSource);

  // 4. Envoi sur Cloudinary
  Logger.log("☁️ Début de l'envoi vers Cloudinary...");
  var timestamp = Math.floor(Date.now() / 1000).toString();
  var dossier = "Accueil/Heros";
  
  var stringToSign = "folder=" + dossier + "&timestamp=" + timestamp + CLOUDINARY_API_SECRET;
  var signatureBytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_1, stringToSign, Utilities.Charset.UTF_8);
  var signature = signatureBytes.map(function(b) { return ("0" + (b < 0 ? b + 256 : b).toString(16)).slice(-2); }).join("");

  var payloadCloudinary = {
    "file": imageSource, 
    "api_key": CLOUDINARY_API_KEY,
    "timestamp": timestamp,
    "signature": signature,
    "folder": dossier
  };

  var optionsCloudinary = {
    "method": "post",
    "payload": payloadCloudinary,
    "muteHttpExceptions": true
  };

  var responseCloudinary = UrlFetchApp.fetch("https://api.cloudinary.com/v1_1/" + CLOUDINARY_CLOUD_NAME + "/image/upload", optionsCloudinary);
  var texteReponseCloudinary = responseCloudinary.getContentText();
  
  var jsonCloudinary;
  try {
    jsonCloudinary = JSON.parse(texteReponseCloudinary);
  } catch (erreur) {
    Logger.log("🚨 ERREUR LECTURE CLOUDINARY : " + texteReponseCloudinary);
    return donnees.urlCloudinary || "";
  }
  
  if (jsonCloudinary.secure_url) {
    Logger.log("🎉 SUCCÈS TOTAL ! Image sauvegardée sur Cloudinary : " + jsonCloudinary.secure_url);
    return jsonCloudinary.secure_url.replace("/upload/", "/upload/q_auto,f_auto/");
  }
  
  Logger.log("🚨 ERREUR CLOUDINARY : " + texteReponseCloudinary);
  return donnees.urlCloudinary || "";
}
